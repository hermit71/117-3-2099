// Регистры Modbus (управление ПЛК стенда)
// Запись в ПЛК

Addr 07: Modbus_CTRL
// Слово управления режимами работы стенда
bits:
    b15 = 0
    b14 = 0 - Ручной режим

    b15 = 0
    b14 = 1 - Автоматический режим (испытания по методике)

    b15 = 1
    b14 = x - Сервисный режим (прямое низкоуровневое управление ПЛК)

    b4...b0 - Управляющая команда для ПЛК (ctrl_command)

    ctrl_command = 0bxxxx0 - Силовое питание серво отключено
                 = 0bxxxx1 - Силовое питание серво включено
                 = 0b00011 - Серво в режиме поддержания заданного момента (с заданной скоростью нарастания момента)
                 = 0b00101 - Jog с заданной скоростью по часовой
                 = 0b01001 - Jog с заданной скоростью против часовой
                 = 0b1xxxx - Сброс ошибки серво
